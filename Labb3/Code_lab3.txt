#include "mbed.h"

// Define pins

Serial pc(USBTX, USBRX);
AnalogIn inSignal(p17);

Ticker timer;
float yf = 0;
float y_old1 = 0;
float y_old2 = 0;
float yf_old1 = 0;
float yf_old2 = 0;

float fir(float y)
{
    yf = 0.0690*y_old2 + 0.8621*y_old1 + 0.0690*y;     
    y_old1 = y;
    y_old2 = y_old1;
    return yf;
}


float butter_low_2(float y)
{
    // Filter time
    yf = 0.978*yf_old2 - 1.978*yf_old1 + 1 + 6.101/100000*y_old2 + 0.000122*y_old1 + 6.101/100000*y;
    y_old1 = y;
    y_old2 = y_old1;
    yf_old1 = yf;
    yf_old2 = yf_old1; 
    return yf;
}

void attime()   // What is done every tick
{
    float y = inSignal*255;
    yf = butter_low_2(y);
    pc.printf("%c", (char)yf);   
}


int main() 
{
    pc.baud(921600);
    timer.attach(&attime, 0.0000125);
    
    while (1)
    {
        
    }
    
}
